```SQL
# 테이블 만들기
CREATE TABLE 이름 (
  테이블구조 INTEGER PRIMARY KEY AUTOINCREMENT,
  테이블이름2 VARCHAR,
  테이블이름3 REAL
);

# FOREIGN KEY 활용
CREATE TABLE orders(
    order_id INTEGER PRIMARY KEY AUTOINCREMENT,
    order_date DATE,
    total_amount REAL,
    customer_id INTEGER,

    CONSTRAINT FK_orders_customers
        FOREIGN KEY (customer_id)
        REFERENCES customers(customer_id)
);

# 테이블 수정하기
UPDATE
  articles
SET
  title = 'update Title'
WHERE
  id = 2;

# 삽입하기
INSERT INTO
  articles (title, content)
VALUES
  ('hello', 'world');

# 조인하기
SELECT
  *
FROM
  articles
INNER JOIN orders
  ON orders.id = articles.id;

# 삭제하기 (서브쿼리 2개혼합)
DELETE FROM
  articles
WHERE id IN (
  SELECT id 
  FROM articles
  ORDER BY 'createdAt'
  LIMIT 2
);

```
